def new_from_csv(csv_row)
  # in gtfs_api models these conventions were use:
  # - remove the inncessary prefix of the column names (ex: agency_name => name)
  # - many files of the feed use an string id, that string is kept in => io_id     
  # - one exception, "route_type" is not kept as "type" because using "type" 
  #   is problematic 
  # 
  # rehash the csv_row, remove the prefix (ex: "agency_" in agency file columns)
  prefix = self.to_s.split("::").last.downcase + '_'
  
  csv_rehashed = Hash[csv_row.map {|k, v| 
    if k.to_s == "#{prefix}id"
      ['io_id'.to_sym, v]
    else #remove prefix except when the resulting string is "type"
      (k.to_s.sub(prefix,'') == 'type')? [k,v] : [k.to_s.sub(prefix,'').to_sym,v] 
    end
    }
  ]
  #puts csv_rehashed
  self.new(csv_rehashed)
end
